Regular Expression II

Implement regular expression matching with support for '.' and '*'.

  1. '.' Matches any single character.
  2. '*' Matches zero or more of the preceding element.
Examples: 
isMatch("aa","a") → 0
isMatch("aa","aa") → 1
isMatch("aaa","aa") → 0
isMatch("aa", "a*") → 1
isMatch("aa", ".*") → 1
isMatch("ab", ".*") → 1
isMatch("aab", "c*a*b") → 1

public class Solution {
    public int isMatch(final String A, final String B) {
        return doesMatch(A, B) ? 1 : 0;
    }
    public boolean doesMatch(String A, String B) {
        if (B.length() == 0) return A.length() == 0;
        // Case 0: First Matches
        boolean firstMatch = (A.length() > 0 && 
            A.charAt(0) == B.charAt(0) || B.charAt(0) == '.');
        
        if (B.length() >= 2 && B.charAt(1) == '*') {
            // Case 1: If first matches
            // Case 2: If not ignore the regex
            return doesMatch(A, B.substring(2)) ||
                (firstMatch && doesMatch(A.substring(1), B));
        }
        return firstMatch && doesMatch(A.substring(1), B.substring(1)); 
    }
}
