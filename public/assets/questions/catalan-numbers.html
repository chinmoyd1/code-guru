    // C(n) = C(0)C(n-1) + C(1)(n-2) .... C(n-1)C(0)
    static int catalanNumber (int n) {
        if (n <= 1) return 1;
        int sum = 0;
        for (int i=0; i<n; i++) {
            sum += catalanNumber(i)*catalanNumber(n-i-1);
        }
        return sum;
    }

    static int catalanNumberDP (int n) {
        int dp[]= new int[n+1];
        dp[0] = 1;
        dp[1] = 1;
        for (int i=2; i<=n; i++) {
            for (int j=0; j<i; j++) {
                dp[i] += dp[j] * dp[i-j-1];
            }
        }
        return dp[n];
    }
